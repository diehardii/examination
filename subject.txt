package seucxxy.csd.examinint.service.impl;

import org.springframework.stereotype.Service;
import seucxxy.csd.examinint.entity.Subject;
import seucxxy.csd.examinint.mapper.SubjectMapper;
import seucxxy.csd.examinint.service.SubjectService;

import java.util.List;

@Service
public class SubjectServiceImpl implements SubjectService {

    private final SubjectMapper subjectMapper;

    public SubjectServiceImpl(SubjectMapper subjectMapper) {
        this.subjectMapper = subjectMapper;
    }

    @Override
    public List<Subject> getAllSubjects() {
        return subjectMapper.findAllSubjects();
    }
    @Override
    public List<String> getSubjectsForUser(String username) {
        return subjectMapper.findSubjectsByUsername(username);
    }

    @Override
    public void addSubject(Subject subject) {
        if (subjectMapper.findSubjectBySubjectName(subject.getSubjectName()) != null) {
            throw new IllegalArgumentException("科目名称已存在");
        }
        subjectMapper.insertSubject(subject);
    }

    @Override
    public void updateSubject(String oldName, Subject newSubject) {
        Subject existingSubject = subjectMapper.findSubjectBySubjectName(oldName);
        if (existingSubject == null) {
            throw new IllegalArgumentException("要修改的科目不存在");
        }

        // 检查新名称是否已被其他科目使用
        Subject subjectWithNewName = subjectMapper.findSubjectBySubjectName(newSubject.getSubjectName());
        if (subjectWithNewName != null && !(subjectWithNewName.getSubjectId()==existingSubject.getSubjectId())) {
            throw new IllegalArgumentException("新科目名称已被其他科目使用");
        }

        subjectMapper.updateSubject(existingSubject.getSubjectId(), newSubject);
    }

    @Override
    public void deleteSubject(String subjectName) {
        Subject subject = subjectMapper.findSubjectBySubjectName(subjectName);
        if (subject == null) {
            throw new IllegalArgumentException("要删除的科目不存在");
        }
        subjectMapper.deleteSubject(subject.getSubjectId());
    }
}

package seucxxy.csd.examinint.controller;

import jakarta.servlet.http.HttpSession;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.*;
import seucxxy.csd.examinint.entity.Subject;
import seucxxy.csd.examinint.entity.User;
import seucxxy.csd.examinint.service.SubjectService;

import java.util.List;

@Controller
@RequestMapping("/exam-paper/subjects")
public class SubjectController {

    private final SubjectService subjectService;

    public SubjectController(SubjectService subjectService) {
        this.subjectService = subjectService;
    }

    @GetMapping
    public String listSubjects(Model model, HttpSession session) {
        User user = (User) session.getAttribute("user");
        List<Subject> subjects = subjectService.getAllSubjects();
        model.addAttribute("subjects", subjects);
        model.addAttribute("user", user);
        return "/exam-paper/subjects";
    }

    @PostMapping
    public String handleSubjectAction(
            @RequestParam("action") String action,
            @RequestParam(value = "subjectName", required = false) String subjectName,
            @RequestParam(value = "oldName", required = false) String oldName,
            @RequestParam(value = "newName", required = false) String newName,
            Model model,
            HttpSession session) {
        User user = (User) session.getAttribute("user");
        model.addAttribute("user", user);

        try {
            switch (action) {
                case "add":
                    subjectService.addSubject(new Subject(subjectName));
                    break;
                case "update":
                    subjectService.updateSubject(oldName, new Subject(newName));
                    break;
                case "delete":
                    subjectService.deleteSubject(subjectName);
                    break;
                default:
                    throw new IllegalArgumentException("无效的操作类型");
            }
            return "redirect:/exam-paper/subjects";
        } catch (Exception e) {
            model.addAttribute("error", e.getMessage());
            return listSubjects(model, session);
        }
    }
}
