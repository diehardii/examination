<script>
import { generateExamPaper } from '@/service/examPaperService';
import { getSubjects } from '@/service/subjectService';
import SideBarMenu from '@/components/SideBarMenu.vue'; // 导入侧边栏组件

export default {
  name: 'PaperGen',
  components: {
    SideBarMenu // 注册组件
  },
  data() {
    return {
      form: {
        name: '',
        content: '',
        difficulty: '',
        subject: '',
        singleChoiceNumber: 10,
        judgeNumber: 5,
      },
      subjects: [],
      error: '',
      success: ''
    };
  },
  async created() {
    try {
      this.subjects = await getSubjects();
    } catch (error) {
      this.error = '获取科目列表失败: ' + error.message;
    }
  },
  methods: {
    async submitForm() {
      this.error = '';
      this.success = '';

      try {
        const paperData = await generateExamPaper(this.form);
        if (paperData?.id) {
          this.$router.push({
            name: 'PaperShow',
            params: { id: paperData.id }
          });
        } else {
          throw new Error('生成的试卷数据不完整，缺少ID');
        }
      } catch (error) {
        this.error = '生成试卷时出错: ' + (error.response?.data?.error || error.message);
      }
    }
  }
};
</script>

      papers: [],

    },
    confirmSelection() {
      if (!this.selectedPaperId) {
        alert('请先选择一份试卷');
        return;
      }
      this.$router.push({ name: 'QuestionAnswer', params: { paperId: this.selectedPaperId } });
    }
  }
};
